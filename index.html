function handleRequest(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sessionsSheet = ss.getSheetByName("sessions"); const answersSheet = ss.getSheetByName("answers"); const action = params.action || ""; if (action === "session") { sessionsSheet.appendRow([ params.session_id || "", params.started_at || "", params.gender || "", params.age_group || "", params.companion_type || "", params.region || "" ]); return ContentService .createTextOutput(JSON.stringify({ ok: true, type: "session" })) .setMimeType(ContentService.MimeType.JSON); } if (action === "answer") { answersSheet.appendRow([ params.session_id || "", params.timestamp || "", params.spot_id || "", params.answer_text || "" ]); return ContentService .createTextOutput(JSON.stringify({ ok: true, type: "answer" })) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({ ok: true, type: "ping" })) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const params = (e && e.parameter) ? e.parameter : {}; return handleRequest(params); } function doPost(e) { const params = (e && e.parameter) ? e.parameter : {}; return handleRequest(params); }
